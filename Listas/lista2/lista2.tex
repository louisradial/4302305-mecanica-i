\begin{lemma}{Trajetória de um potencial central}{trajetória}
    Um sistema com massa reduzida \(\mu\) submetido a um potencial central \(U = U(r)\) tem trajetória definida pela integração da relação
    \begin{equation}
        \frac{L}{r^2\sqrt{2\mu\left(E-U(r)-\frac{L^2}{2\mu r^2}\right)}}\diff{r}{\theta} = \pm1,
    \end{equation}
    onde \(L = \mu r^2 \dot\theta\) é o momento angular e \(E\) é a energia do sistema.
\end{lemma}
\begin{proof}
    Segue da conservação do momento angular em sistemas com potencial central que
    \begin{equation}
        \diff{\theta}{t} = \frac{L}{\mu r^2},
    \end{equation}
    ou então, ao multiplicar \(\diff{t}{\theta}\) em ambos os lados da equação e utilizando a regra da cadeia,
    \begin{equation}
        \diff{t}{\theta} = \frac{\mu r^2}{L}.
    \end{equation}
    Pela conservação da energia do sistema, temos
    \begin{equation}
        \frac12 \mu\inner*{\diff{\vec{r}}{t}}{\diff{\vec{r}}{t}} + U(r) = E.
    \end{equation}

    Em coordenadas polares, temos \(\diff{\vec{e}_r}{t} = \diff{\theta}{t}\vec{e}_\theta\), portanto
    \begin{align}
        \inner*{\diff{\vec{r}}{t}}{\diff{\vec{r}}{t}} &= \inner*{\dot{r} \vec{e}_r + r\dot{\theta} \vec{e}_\theta}{\dot{r} \vec{e}_r + r\dot{\theta} \vec{e}_\theta}\\
                                                      &= \dot{r}^2 + \left(r\dot{\theta}\right)^2\\
                                                      &= \dot{r}^2 + r^2\left(\frac{L}{\mu r^2}\right)^2\\
                                                      &= \dot{r}^2 + \frac{L^2}{\mu^2r^2}.
    \end{align}
    Assim, a equação da energia do sistema pode ser escrita como
    \begin{equation}
        E = \frac12 \mu \dot{r}^2 + \frac{L^2}{2\mu r^2} + U(r).
    \end{equation}
    Isolando \(\dot{r}\), obtemos
    \begin{equation}
        \diff{r}{t} = \pm \sqrt{\frac{2}{\mu}} \sqrt{E - U(r) - \frac{L^2}{2\mu r^2}}.
    \end{equation}
    Multiplicando ambos os lados por \(\diff{t}{\theta}\) e utilizando a regra da cadeia, obtemos
    \begin{align}
        \diff{r}{\theta} &= \pm \sqrt{\frac{2}{\mu}} \sqrt{E - U(r) - \frac{L^2}{2\mu r^2}} \diff{t}{\theta}\\
                         &=  \pm \sqrt{2\mu\left(E - U(r) - \frac{L^2}{2\mu r^2}\right)} \frac{r^2}{L}.
    \end{align}
    Dividindo ambos os lados pelo lado direito, obtemos a expressão dada no \cref{lem:trajetória}.
\end{proof}
